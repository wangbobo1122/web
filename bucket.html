\begin{algorithm}
            \begin{algorithmic}
            \FUNCTION{bucketsort}{$A$}
                \STATE $epoch⬅0,max⬅MAX(A),temp⬅max\hspace{10px}//MAX()求数组中最大元素$
                \WHILE{$temp > 0$}
                    \STATE $temp⬅ \lfloor temp/10 \rfloor$
                    \STATE $epoch⬅epoch+1$
                \ENDWHILE

                \For{$i=0$ to $epoch$}
                    \STATE $B⬅[[\hspace{5px}],...,[\hspace{5px}]]\hspace{10px}//定义10个二维数组$
                    \For{$j=0$ to $A.length$}
                        \STATE$index⬅ \lfloor A[j]/(10^i) \rfloor \%10 $
                        \STATE$B[index].push(A[j])$
                    \EndFor
                    \STATE$A=FLAT(B)\hspace{10px}//将二位数组展开成一维数组$
                \EndFor

                \Return A
            \ENDFUNCTION
            \end{algorithmic}  
            \end{algorithm}
