\begin{algorithm}
            \begin{algorithmic}

            \FUNCTION{maxheapify}{$A,i,j$}
                \STATE $father⬅i,child⬅father*2+1$
                \WHILE{$child <= j$}
                    \IF{$child+1 <= j \&\& A[child+1]>A[child]$}
                        \STATE $child⬅child+1$
                    \ENDIF
                    \IF{$A[father]>A[child]$}
                        \STATE $return$
                    \ELSE
                        \STATE $swap(A[father],A[child])$
                        \STATE $father⬅child,child⬅father*2+1$
                    \ENDIF    
                \ENDWHILE
                \Return 
            \ENDFUNCTION

            \FUNCTION{heapsort}{$A$}
                \STATE $len⬅A.length$
                \FOR{$i⬅len/2-1$ \TO $0$}  
                    \STATE $maxheapify(A,i,len-1)$
                \ENDFOR
                \FOR{$i⬅len-1$ \TO $1$}  
                    \STATE $swap(A[0],A[i])$
                    \STATE $maxheapify(A,0,i-1)$
                \ENDFOR
                \Return 
            \ENDFUNCTION

            \end{algorithmic}  
            \end{algorithm}
